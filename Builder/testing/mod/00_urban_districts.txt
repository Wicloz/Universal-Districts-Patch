			weight = 20000

			owner = {

			}
			has_global_flag = gai_enabled_flag
		}
		modifier = {
			factor = 0
			OR = {

					value > 0

					value > 0

					type = district_farming
					value > 0

				gai_focus_world = yes
			}

			free_housing > 1
			has_global_flag = gai_enabled_flag
		}
	}
	ai_resource_production = {
		consumer_goods = 1
		trigger = {
			always = yes
		}
	}
