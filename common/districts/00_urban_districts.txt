	base_buildtime = 480
	is_capped_by_modifier = no

	show_on_uncolonized = {
		exists = from
		from = { is_regular_empire = yes }
		NOT = { is_planet_class = pc_habitat }
		NOT = { is_planet_class = pc_city }
		NOT = { is_planet_class = pc_ecocity }
		NOT = { is_planet_class = pc_archive }
		NOT = { is_planet_class = pc_mem_vazuran_habitat }
		NOT = { is_planet_class = pc_dyson_swarm }
		NOT = { is_planet_class = pc_habitable_asteroid }
		NOT = { is_planet_class = pc_asteroid_base }
		NOT = { is_planet_class = pc_moon_base }
		NOT = { has_modifier = pm_Utopian_Paradise }
		NAND = {
			is_planet_class = pc_ringworld_habitable
			has_global_flag = lrsk_megastrcutre_district_rework_mod_active
		}
	}

	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
		NOT = { is_planet_class = pc_habitat }
		NOT = { is_planet_class = pc_city }
		NOT = { is_planet_class = pc_ecocity }
		NOT = { is_planet_class = pc_archive }
		NOT = { is_planet_class = pc_mem_vazuran_habitat }

		NAND = {
			is_planet_class = pc_ringworld_habitable
			has_global_flag = lrsk_megastrcutre_district_rework_mod_active
		}
	}

	allow = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
	}

	conversion_ratio = 0.5
	convert_to = {
		district_arcology_housing
		district_nexus
		district_hive
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = @city_cost
		}
		upkeep = {
			energy = 2
		}
	}

	planet_modifier = {
		planet_housing_add = 5
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_keepers_harmony
			}
		}
		modifier = {
			planet_housing_add = -1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = -1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				is_fallen_empire = no
			}
		}
		modifier = {
			job_clerk_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				is_fallen_empire = no
				or = {
					has_policy_flag = state_education_preadult
					has_policy_flag = state_education_postadult
				}
			}
		}
		modifier = {
			job_teacher_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				is_fallen_empire = no
				has_policy_flag = alp_private_businesses_prohibited
			}
		}
		modifier = {
			job_service_add = 1
			job_labourer_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				is_fallen_empire = no
				has_policy_flag = alp_private_businesses_regulated

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				is_fallen_empire = no

		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_non_swapped_tradition = tr_prosperity_public_works }
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_1
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_2
				NOR = {
					has_valid_civic = civic_keepers_harmony
					has_valid_civic = civic_agrarian_idyll
				}
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_non_swapped_tradition = tr_prosperity_interstellar_franchising }
		}
		modifier = {
			job_clerk_add = 1
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		text = job_clerk_effect_desc
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 1
			NOT = { has_global_flag = gai_enabled_flag }
		}
		# GAI #
		modifier = {
			weight = 10000
			planet = {
				planet_devastation < 1
				years_passed > 10
				count_pops > 6
				free_housing < 4
				NOT = {
					any_owned_pop = {
						OR = {
							is_unemployed = yes
							has_job = servant
						}
					}
				}
			}
			has_global_flag = gai_enabled_flag
		}
		modifier = {
			weight = 80
			planet = {
				planet_devastation < 1
				gai_good_research = yes
				free_housing < 6
			}
			has_global_flag = gai_enabled_flag
		}
		modifier = {
			weight = 50
			factor = 1
			num_free_districts = { type = district_generator value = 0 }
			num_free_districts = { type = district_mining value = 0 }
			num_free_districts = { type = district_farming value = 0 }
			num_free_districts = { type = district_generator_uncapped value = 0 }
			num_free_districts = { type = district_mining_uncapped value = 0 }
			num_free_districts = { type = district_farming_uncapped value = 0 }
			NOT = { any_deposit = {	is_blocker = yes } }
			has_global_flag = gai_enabled_flag
		}
		# VANILLA #
		modifier = {
			factor = 2
			owner = {
				has_monthly_income = {
					resource = consumer_goods
					value < 0
				}
			}
			NOT = { has_global_flag = gai_enabled_flag }
		}
		modifier = {
			factor = 5
			free_housing < 0
			NOT = { has_global_flag = gai_enabled_flag }
		}
		modifier = {
			factor = 0
			free_housing > 2
			NOT = { has_global_flag = gai_enabled_flag }
		}
	}

	ai_resource_production = {
		consumer_goods = 1
		trigger = {
			always = yes
		}
	}
}
