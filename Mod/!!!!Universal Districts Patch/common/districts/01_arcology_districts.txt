district_arcology_housing = {
	base_buildtime = 600
	is_capped_by_modifier = no

	show_on_uncolonized = {
		or = {
			is_planet_class = pc_city
			is_planet_class = pc_ecocity
		}
	}

	potential = {
		or = {
			is_planet_class = pc_city
			is_planet_class = pc_ecocity
		}
	}

	conversion_ratio = 1
	convert_to = {
		district_city
		district_nexus
		district_hive
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
		}
		upkeep = {
			energy = 5
		}
	}

	planet_modifier = {
		planet_housing_add = 15
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_maintenance_drone_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_fallen_empire = no
				is_regular_empire = yes
			}
		}
		modifier = {
			job_clerk_add = 5
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				is_fallen_empire = no
				or = {
					has_policy_flag = state_education_preadult
					has_policy_flag = state_education_postadult
				}
			}
		}
		modifier = {
			job_teacher_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				is_fallen_empire = no
				has_policy_flag = alp_private_businesses_prohibited
			}
		}
		modifier = {
			job_service_add = 2
			job_labourer_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				is_fallen_empire = no
				has_policy_flag = alp_private_businesses_regulated
			}
		}
		modifier = {
			job_retailer_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				is_fallen_empire = no
				has_policy_flag = alp_private_businesses_freedom
			}
		}
		modifier = {
			job_retailer_add = 3
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_fallen_empire = no
				is_regular_empire = yes
			}
		}
		text = job_clerk_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_maintenance_drone_effect_desc
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 1
			NOT = {
				has_global_flag = gai_enabled_flag
			}
		}
		modifier = {
			weight = 5
			free_housing < 10
			has_global_flag = gai_enabled_flag
		}
	}
}
